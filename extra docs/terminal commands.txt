  SSH Access

  # Exit EC2
  exit
  # or press Ctrl+D

  Docker Services (Primary Method)

  # Navigate to project
  cd ~/nifty_options_agent

  # View all container status
  docker-compose ps

  # Start all services
  docker-compose up -d

  # Stop all services
  docker-compose down

  # Restart all services
  docker-compose restart

  # Restart single service
  docker-compose restart trading_agent
  docker-compose restart openalgo
  docker-compose restart monitor_dashboard

  View Logs

  # Live logs (all services)
  docker-compose logs -f

  # Live logs (specific service)
  docker-compose logs -f trading_agent
  docker-compose logs -f openalgo
  docker-compose logs -f monitor_dashboard

  # Last 100 lines
  docker-compose logs --tail=100 trading_agent

  # Exit live logs
  # Press Ctrl+C

  Database Access (EC2)

  # Access SQLite database
  cd ~/nifty_options_agent
  docker-compose exec trading_agent sqlite3 /app/baseline_v1_live/live_state.db

  # Quick queries inside sqlite3
  .tables                                    # List all tables
  SELECT * FROM positions;                   # View positions
  SELECT * FROM daily_summary ORDER BY date DESC LIMIT 5;
  .quit                                      # Exit sqlite3

  System Monitoring

  # Check disk space
  df -h

  # Check memory usage
  free -h

  # Check running processes
  htop
  # Press 'q' to quit

  # Check Docker resource usage
  docker stats
  # Press Ctrl+C to exit

  Nginx (Web Server)

  # Check nginx status
  sudo systemctl status nginx

  # Restart nginx
  sudo systemctl restart nginx

  # View nginx logs
  sudo tail -f /var/log/nginx/access.log
  sudo tail -f /var/log/nginx/error.log

  # Test nginx configuration
  sudo nginx -t

  Emergency Commands

  Kill Hung Processes (Local)

  # Find Python processes
  Get-Process python

  # Kill specific process by PID
  Stop-Process -Id <PID> -Force

  Emergency Stop (EC2)

  # Stop all containers immediately
  cd ~/nifty_options_agent
  docker-compose down

  # Force remove containers
  docker-compose down --remove-orphans

  # Nuclear option (stop all Docker containers)
  docker stop $(docker ps -q)

  Check Service Health (EC2)

  # Check if trading agent is running
  docker-compose ps | grep trading_agent

  # Check last restart time
  docker inspect trading_agent | grep StartedAt

  # View container resource usage
  docker stats trading_agent --no-stream
